posterDetailResolver
Type : ResolveFn<DetailPoster | null>
Default value : (route) => { const { mediaType, id } = route.params; const theMovieDBPort = inject(TheMovieDBPort); switch (mediaType) { case 'movie': return theMovieDBPort.getMovieDetails(id).pipe(map((movie) => movie as DetailPoster)); case 'tv': return theMovieDBPort.getSeriesDetails(id).pipe(map((series) => series as DetailPoster)); default: return null; } }
